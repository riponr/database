CREATE TABLE productlist
(
ID NUMBER   NOT NULL ,
NAME VARCHAR2(4000),
PRICE VARCHAR2(4000),
STATUS VARCHAR2(4000)
);

ALTER TABLE productlist ADD (
  CONSTRAINT prli_pk PRIMARY KEY (ID));
  
  CREATE SEQUENCE prli_seq START WITH 101;
  
  CREATE OR REPLACE TRIGGER prli_bir 
BEFORE INSERT ON productlist 
FOR EACH ROW
BEGIN
  SELECT prli_seq.NEXTVAL
  INTO   :new.id
  FROM   dual;
END;
