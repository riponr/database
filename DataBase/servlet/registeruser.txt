CREATE TABLE registeruser
(
ID NUMBER   NOT NULL ,
NAME VARCHAR2(4000),
PASSWORD VARCHAR2(4000),
EMAIL VARCHAR2(4000),
COUNTRY VARCHAR2(4000)
);

ALTER TABLE registeruser ADD (
  CONSTRAINT reus_pk PRIMARY KEY (ID));
  
  CREATE SEQUENCE reus_seq START WITH 101;
  
  CREATE OR REPLACE TRIGGER reus_bir 
BEFORE INSERT ON registeruser 
FOR EACH ROW
BEGIN
  SELECT reus_seq.NEXTVAL
  INTO   :new.id
  FROM   dual;
END;
