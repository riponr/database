CREATE TABLE emregister
(
ID NUMBER   NOT NULL ,
NAME VARCHAR2(4000),
PASSWORD VARCHAR2(4000),
EMAIL VARCHAR2(4000),
COUNTRY VARCHAR2(4000)
);

ALTER TABLE emregister ADD (
  CONSTRAINT emr_pk PRIMARY KEY (ID));
  
  CREATE SEQUENCE emr_seq START WITH 1001;
  
  CREATE OR REPLACE TRIGGER emre_bir 
  BEFORE INSERT ON emregister 
  FOR EACH ROW
  BEGIN
  SELECT emr_seq.NEXTVAL
  INTO   :new.id
  FROM   dual;
END;
